# WeeChat settings
# By Henrik Lissner <henrik@lissner.net>
#
# Pass this file to `./bin/wee` while weechat is running.

/mouse enable

/set irc.server_default.ssl on
/set irc.server_default.ssl_verify on
/set irc.look.server_buffer independent



########################################
#
# Plugins
#
########################################

/script install iset.pl
/script install buffers.pl
/script install go.py

/script install colorize_lines.pl
/set plugins.var.perl.colorize_lines.lines on

/script install colorize_nicks.py
/set colorize_nicks.look.colorize_input on
/set colorize_nicks.look.ignore_nicks_in_urls on

/script install pushover.pl
/set plugins.var.ruby.pushover-weechat.userkey "${sec.data.pushover_key}"



########################################
#
# Looks
#
########################################

# One single input bar for all windows
/bar add rootinput root bottom 1 0 buffer_name+[input_prompt]+(away),[input_search],[input_paste],> input_text
/bar del input

# Statusbar
/set weechat.bar.status.items [buffer_last_number],[buffer_plugin],buffer_number+:+buffer_name+{buffer_nicklist_count}+buffer_zoom+buffer_filter,scroll,[lag],[hotlist],completion

# Buffer list
/set weechat.bar.buffers.position left
#/set weechat.bar.buffers.filling_top_bottom horizontal
/set buffers.look.prefix on
/set buffers.look.hide_merged_buffers all
/set buffers.color.none_channel_fg 244
/set buffers.color.current_bg blue
/set buffers.color.current_fg default

/set weechat.look.prefix_same_nick "*"
/set weechat.bar.nicklist.size 15
/set weechat.bar.buffers.size 20
/set weechat.color.chat_channel 247
/set irc.look.color_nicks_in_nicklist on
/set irc.look.color_nicks_in_server_messages off

# greyed out server messages
/set irc.color.message_join 243
/set irc.color.message_quit 243
/set irc.color.notice 243
/set irc.color.reason_quit 243
/set irc.color.topic_current 243
/set irc.color.topic_new 243
/set irc.color.topic_old 243
/set weechat.color.chat_host 243
/set weechat.color.chat_nick 243
/set weechat.color.chat_nick_self *white
/set weechat.color.chat_delimiters 243
/set weechat.color.chat_time_delimiters 243
/set logger.color.backlog.end lightgreen
/set logger.color.backlog.line 243
/set weechat.color.nicklist_away 243


###
/save
/print Weechat configured!
